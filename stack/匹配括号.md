# 匹配括号

```
class Solution {
    public boolean isValid(String s) {
        LinkedList<Character> stk=new LinkedList<>();
                
        // for (int i = 0; i < s.length(); i++) {
        for(int i=0;i<s.length();i++){
            char c = s.charAt(i);
           if(c=='{'||c=='('||c=='['){
            stk.push(c);
           }else{
                if(stk.isEmpty())return false;
                else {
                    char top = stk.pop();
                    if(c==')' && top!='(') return false;
                    if(c==']' && top!='[') return false;
                    if(c=='}' && top!='{') return false;
                }
           }
        }
        if(stk.isEmpty())
        return true;
        else return false;
    }
}
```
