# 环形链表（判断链表是不是有环）

/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle(ListNode head) {
        // HashSet<ListNode> h1=new HashSet<>();
        // while(head!=null){
        //     if(h1.contains(head)){
        //         return true;
        //     }
        //     h1.add(head);
        //     head=head.next;
        // }
        // return false;
        ListNode pre=head,cnt=head;
        while(cnt!=null&&cnt.next!=null){ 
            pre=pre.next;
            cnt=cnt.next.next;
            if(pre==cnt)return true;
        }
        return false;

    }
}
